382 -- [[ Configure LSP ]]
383 --  This function gets run when an LSP connects to a particular buffer.
384 local on_attach = function(_, bufnr)
385     -- NOTE: Remember that lua is a real programming language, and as such it is possible
386     -- to define small helper and utility functions so you don't have to repeat yourself
387     -- many times.
388     --
389     -- In this case, we create a function that lets us more easily define mappings specific
390     -- for LSP related items. It sets the mode, buffer and description for us each time.
391     local nmap = function(keys, func, desc)
392         if desc then
393             desc = 'LSP: ' .. desc
394         end
395 
396         vim.keymap.set('n', keys, func, { buffer = bufnr, desc = desc })
397     end
398 
399     nmap('<leader>rn', vim.lsp.buf.rename, '[R]e[n]ame')
400     nmap('<leader>ca', vim.lsp.buf.code_action, '[C]ode [A]ction')
401 
402     nmap('gd', require('telescope.builtin').lsp_definitions, '[G]oto [D]efinition')
403     nmap('gr', require('telescope.builtin').lsp_references, '[G]oto [R]eferences')
404     nmap('gI', require('telescope.builtin').lsp_implementations, '[G]oto [I]mplementation')
405     nmap('<leader>D', require('telescope.builtin').lsp_type_definitions, 'Type [D]efinition')
406     nmap('<leader>ds', require('telescope.builtin').lsp_document_symbols, '[D]ocument [S]ymbols')
407     nmap('<leader>ws', require('telescope.builtin').lsp_dynamic_workspace_symbols, '[W]orkspace [S]ymbols')
408 
409     -- See `:help K` for why this keymap
410     nmap('K', vim.lsp.buf.hover, 'Hover Documentation')
411     nmap('<C-k>', vim.lsp.buf.signature_help, 'Signature Documentation')
412 
413     -- Lesser used LSP functionality
414     nmap('gD', vim.lsp.buf.declaration, '[G]oto [D]eclaration')
415     nmap('<leader>wa', vim.lsp.buf.add_workspace_folder, '[W]orkspace [A]dd Folder')
416     nmap('<leader>wr', vim.lsp.buf.remove_workspace_folder, '[W]orkspace [R]emove Folder')
417     nmap('<leader>wl', function()
418         print(vim.inspect(vim.lsp.buf.list_workspace_folders()))
419     end, '[W]orkspace [L]ist Folders')
420 
421     -- Create a command `:Format` local to the LSP buffer
422     vim.api.nvim_buf_create_user_command(bufnr, 'Format', function(_)
423         vim.lsp.buf.format()
424     end, { desc = 'Format current buffer with LSP' })
425 end
